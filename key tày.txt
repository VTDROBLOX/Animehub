-- // Hàm kiểm tra key
local function checkKey(inputKey)
    inputKey = inputKey:gsub("%s+", "") -- bỏ khoảng trắng
    for _, validKey in ipairs(VALID_KEYS) do
        if inputKey == validKey then
            return true
        end
    end
    return false
end

-- // Hàm mở hub sau khi xác minh key
local function onKeySuccess()
    StatusLabel.Text = "✓ Key hop le! Dang mo hub..."
    StatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
    
    task.wait(0.8)
    
    -- Animate card ra
    tween(Card, {Position = UDim2.new(0.5, -190, -0.5, -260), BackgroundTransparency = 1}, 0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In)
    tween(Overlay, {BackgroundTransparency = 1}, 0.5)
    
    task.wait(0.6)
    ScreenGui:Destroy()
    
    -- =============================================
    -- // LOAD HUB CỦA BẠN Ở ĐÂY
    loadstring(game:HttpGet("https://raw.githubusercontent.com/VTDROBLOX/Animehub/refs/heads/main/Tayhubbnana.txt"))()
    -- =============================================
end

-- // Xác minh key khi nhấn nút
VerifyBtn.MouseButton1Click:Connect(function()
    local input = KeyInput.Text
    if input == "" then
        StatusLabel.Text = "✗ Vui long nhap key!"
        StatusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return
    end
    
    if checkKey(input) then
        onKeySuccess()
    else
        StatusLabel.Text = "✗ Key khong hop le! Thu lai hoac lay key moi."
        StatusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        tween(InputRow, {Position = UDim2.new(0, 25, 0, 224)}, 0.05)
        task.wait(0.05)
        tween(InputRow, {Position = UDim2.new(0, 15, 0, 224)}, 0.05)
        task.wait(0.05)
        tween(InputRow, {Position = UDim2.new(0, 20, 0, 224)}, 0.05)
    end
end)

-- // Cho phép nhấn Enter để xác minh

    if enterPressed then
        VerifyBtn.MouseButton1Click:Fire()
    end
end)
